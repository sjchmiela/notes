    \textbf{Dane:}\\
    $P(A\cap B) = \frac13, P(A\cup B) = \frac56, P(B') = \frac12$.\\
    \textbf{Szukane:}\\
    $P(A), P(B), P(A|B)$.\\
    \textbf{Rozwiązanie:}\\
    $P(B) = 1-P(B') = 1-\frac12 = \mathbf{\frac12}$\\
    $P(A) = P(A\cup B) - P(B) + P(A\cap B) = \frac56 - \frac12 + \frac13 = \mathbf{\frac23}$\\
    $P(A|B) = \frac{P(A\cap B)}{P(B)} = \frac{\frac13}{\frac12} = \mathbf{\frac23}$
    \end{minipage}
};
